# UniversalSection

### Props

```js 
type TItem = {
	icon: string;
	text: string;
}

type TImage = {
	url: string;
	title?: string;
	description?: string;
}

interface Props {
	section: {
		id?: string;
		sectionClass?: string;
		items?: TItem[];
		suptitle?: string;
		title?: string;
		text?: string;
		images?: TImage[];
		button?: string;
	}
}
```
